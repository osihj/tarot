import { useMemo } from "react";

import AppHeader from "@/components/AppHeader";

import GreetingCard from "@/components/GreetingCard";

import WeeklySummary from "@/components/WeeklySummary";

import HabitTracker from "@/components/HabitTracker";

import CheckInButton from "@/components/CheckInButton";

import BottomNav from "@/components/BottomNav";

import { toast } from "sonner";


const getWeekDays = () => {

  const today = new Date();

  const dayOfWeek = today.getDay();

  const monday = new Date(today);

  monday.setDate(today.getDate() - ((dayOfWeek + 6) % 7));


  return Array.from({ length: 7 }, (_, i) => {

    const d = new Date(monday);

    d.setDate(monday.getDate() + i);

    return {

      date: d.getDate(),

      completed: false,

      isToday: d.toDateString() === today.toDateString(),

    };

  });

};


const Index = () => {

  const weekDays = useMemo(() => getWeekDays(), []);


  const handleCheckIn = () => {

    toast.success("打卡成功！繼續保持 💪");

  };


  return (

    <div className="max-w-md mx-auto min-h-screen bg-background relative flex flex-col">

      <AppHeader />

      <main className="flex-1 p-4 pb-32">

        <div className="flex flex-col space-y-2">

          <GreetingCard name="自律君" weeklyCheckins={0} rank="加油！動起來" />

          <WeeklySummary totalMinutes={0} />

          <HabitTracker title="練功" count={0} days={weekDays} />

          <HabitTracker title="五分鐘" count={0} days={weekDays} />

          <CheckInButton onClick={handleCheckIn} />

        </div>

      </main>

      <BottomNav />

    </div>

  );

};


export default Index;
